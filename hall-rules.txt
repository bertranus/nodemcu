on hall#State do
  gpio,15,[hall#State]
endon

//сброс счетчика по достижении лимита
on counter#Total do
  if [counter#Total]>=[arg#maxCount]
    ResetPulseCounter
  endif
endon

// установка максимума оборотов
on System#Boot do
  ResetPulseCounter
  TaskValueSet arg,1,5
  TaskValueSet arg,2,15
endon
